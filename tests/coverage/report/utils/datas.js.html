/**
* @ngdoc service
* @name com.ionic.quickstart.utils.data:data
* @author Fábio Pereira <fabio.pereira.gti@gmail.com>
* @description
* Servico que verifica a validade das datas inseridas no dispositivo
*/
(function () {
  'use strict';
  angular.module('com.ionic.quickstart.utils.data', [])
  .service('Data', [function () {
 
    /**
    * @ngdoc property
    * @name com.ionic.quickstart.utils.data:data#self
    * @propertyOf com.ionic.quickstart.utils.data:data
    * @description
    * variavel proxy
    */
    var self = this;
 
 
    /**
    * @ngdoc method
    * @name com.ionic.quickstart.utils.data:data#Date.prototype.pt_br
    * @methodOf com.ionic.quickstart.utils.data:data
    * @description
    * Método que adiciona ao objeto Date do javascript funcionalidade de conversão de uma String no
      padrão pt-br em um Object<Date>
    * @param {Strind} data - String no formato dd/mm/aaaa a ser convertida em Date
    * @returns {Date} Data referente a string informada
    */
    self.pt_br = function (data) {
      var array = data.split('/');
      var dia = array[0];
      var mes = array[1];
      var ano = array[2];
      return new Date(ano.concat('/', mes, '/', dia));
    };
 
  }]);
})();
